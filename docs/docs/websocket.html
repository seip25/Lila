WebSocket System (Real-Time)

Lila Framework provides a zero-dependency, Socket.io-style WebSocket engine built on top of Starlette. It features binary orjson serialization on the server, room management, automatic heartbeat ping/pong, and a smart reconnecting client (public/js/ws.js).

1. Backend Setup (lila/core/websocket.py)

Register WebSocket routes with @router.websocket("/ws") and handle events using WebSocketManager:

app/routes/web/chat.py 

from lila.core.routing import Router
from lila.core.websocket import WebSocketManager, WebSocket
from lila.core.security import Security
import datetime

router = Router()
ws_manager = WebSocketManager()

# Event handler for joining a room
@ws_manager.on("join_channel")
async def handle_join(websocket: WebSocket, data: dict):
    room = data.get("room", "general")
    username = Security.sanitize_data(data.get("username", "Guest"))
    
    ws_manager.join_room(websocket, room)
    
    # Broadcast join notice to room members
    await ws_manager.broadcast_to_room(room, "user_joined", {
        "user": username,
        "room": room,
        "time": datetime.datetime.now().strftime("%H:%M")
    })

# Event handler for sending chat messages
@ws_manager.on("send_message")
async def handle_message(websocket: WebSocket, data: dict):
    room = data.get("room", "general")
    text = Security.sanitize_data(data.get("text", ""))
    username = Security.sanitize_data(data.get("username", "Guest"))
    
    if not text.strip():
        return

    # Broadcast message to room subscribers
    await ws_manager.broadcast_to_room(room, "new_message", {
        "user": username,
        "text": text,
        "room": room,
        "time": datetime.datetime.now().strftime("%H:%M")
    })

# Register WebSocket endpoint
@router.websocket("/ws/chat")
async def chat_endpoint(websocket: WebSocket):
    await ws_manager.handle_connection(websocket)

2. Client-Side Usage (public/js/ws.js)

Connect from Jinja2 templates using the built-in lilaWS client:

resources/html/chat.jinja 

<!DOCTYPE html>
<html lang="en">
<head>
  <title>Real-Time Chat</title>
  {{ public('js/tailwind.js') | safe }}
  <script src="{{ public('js/ws.js') }}"></script>
</head>
<body class="bg-slate-900 text-slate-100 p-6 font-sans flex justify-center">
  <div class="w-full max-w-xl bg-slate-800 border border-slate-700 rounded-2xl p-6 shadow-2xl space-y-4">
    <div class="flex items-center justify-between border-b border-slate-700 pb-3">
      <h3 class="font-bold">Chat Room (<span id="room-label" class="text-blue-400">#general</span>)</h3>
      <span id="status-indicator" class="text-xs px-2 py-1 rounded bg-amber-500/20 text-amber-400 font-mono">Connecting...</span>
    </div>

    <div id="messages" class="h-64 overflow-y-auto bg-slate-950 p-4 rounded-xl space-y-2 text-sm"></div>

    <form onsubmit="sendMessage(event)" class="flex gap-2">
      <input type="text" id="msg" placeholder="Type a message..." class="flex-1 bg-slate-900 border border-slate-700 rounded-xl px-4 py-2" required />
      <button type="submit" class="bg-blue-600 hover:bg-blue-500 text-white font-semibold px-4 py-2 rounded-xl">Send</button>
    </form>
  </div>

  <script>
    const ws = lilaWS('ws://' + location.host + '/ws/chat');

    ws.on('connect', () => {
      document.getElementById('status-indicator').className = 'text-xs px-2 py-1 rounded bg-emerald-500/20 text-emerald-400 font-mono';
      document.getElementById('status-indicator').innerText = 'Connected';
      ws.emit('join_channel', { room: 'general', username: 'Alice' });
    });

    ws.on('disconnect', () => {
      document.getElementById('status-indicator').className = 'text-xs px-2 py-1 rounded bg-rose-500/20 text-rose-400 font-mono';
      document.getElementById('status-indicator').innerText = 'Disconnected';
    });

    ws.on('new_message', (data) => {
      const box = document.getElementById('messages');
      box.innerHTML += `<div><b class="text-blue-400">${data.user}</b> <span class="text-xs text-slate-500">[${data.time}]</span>: ${data.text}</div>`;
      box.scrollTop = box.scrollHeight;
    });

    function sendMessage(e) {
      e.preventDefault();
      const input = document.getElementById('msg');
      ws.emit('send_message', { room: 'general', username: 'Alice', text: input.value });
      input.value = '';
    }
  </script>
</body>
</html>

3. API Reference Summary

Below are the primary methods for backend and client WebSocket management:

API Methods 

# Backend (WebSocketManager)
@ws_manager.on("event_name")                            # Register event callback
await ws_manager.broadcast("event", data)               # Send event to all sockets
await ws_manager.broadcast_to_room("room", "event", data)# Send event to room members
await ws_manager.emit_to(websocket, "event", data)      # Send event to single socket
ws_manager.join_room(websocket, "room_name")           # Join socket to channel
ws_manager.leave_room(websocket, "room_name")          # Remove socket from channel

# Client JavaScript (lilaWS)
const ws = lilaWS('ws://domain/ws')                    # Connect to server
ws.on('event', (data) => { ... })                      # Handle server event
ws.emit('event', { ... })                              # Send event to server
ws.join('room_name')                                   # Join channel
ws.leave('room_name')                                  # Leave channel
